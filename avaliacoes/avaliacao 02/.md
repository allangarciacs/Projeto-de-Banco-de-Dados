-- 1) Liste todos os autores que escrevem livros do gênero Infantil.
select distinct Autor.nome
from Autor, Livro_Autor, Livro, Genero
where Autor.idAutor = Livro_Autor.idAutor and
	  Livro_Autor.idLivro = Livro.idLivro and
	  Livro.idGenero = Genero.idGenero 	  and
      Genero.descricao = 'Infantil';

/*
nome
Pedro
Francisco
Felipe
Maria
*/

-- 2) Liste todos os livros do gênero Ficção, seus respectivos autores e suas respectivas editoras
-- (nomes de editoras).
select Livro.titulo, Autor.nome as Autor, Editora.nome as Editora
from Livro, Autor, Editora, Genero, Livro_Autor
where Livro.idGenero = Genero.idGenero    and 
	  Livro.idEditora = Editora.idEditora and 
      Livro.idLivro = Livro_Autor.idLivro and
      Livro_Autor.idAutor = Autor.idAutor and Genero.descricao = 'Ficção';

/*
titulo    Autor       Editora
C	        Pedro	      Scipione
C	        Ana	        Scipione
C	        Maria	      Scipione
I	        Francisco	  Makron Books
*/

-- 3) Liste quantos livros o autor Pedro já publicou.
select Autor.nome, count(Livro_Autor.idLivro) as QuantidadeLivros
from Autor, Livro_Autor
where Autor.idAutor = Livro_Autor.idAutor and
      Autor.nome = 'Pedro';

/*
nome      QuantidadeLivros
Pedro	    4
*/

-- 4) Mostre as editoras que publicam livros do gênero Negócio.
select distinct Editora.nome
from Editora, Livro, Genero
where Editora.idEditora = Livro.idEditora and
	  Livro.idGenero = Genero.idGenero 	  and
      Genero.descricao = 'Negócios';

/*

*/

-- 5) Liste os autores que publicaram pela editora Sagra-Luzato.

-- EXTRA quais  os livros que tem mais de um autor
select livro.titulo, autor.nome
from Autor, Livro, Livro_Autor
where livro.idLivro = Livro_Autor.idLivro and 
	  autor.idAutor = Livro_Autor.idAutor and 
      livro.idLivro in (
		select idLivro
		from livro_autor
		group by idLivro
		having count(idAutor) > 1
	);
    
 -- VERSÃO COM JOIN    
SELECT a.nome, l.titulo
FROM livro l
JOIN livro_autor la_principal ON l.idLivro = la_principal.idLivro
JOIN autor a ON a.idAutor = la_principal.idAutor
WHERE l.idLivro IN (
    SELECT idLivro 
    FROM livro_autor 
    GROUP BY idLivro 
    HAVING COUNT(idAutor) > 1
);
